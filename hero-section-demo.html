CSS Implementation

.hero-section {
  background: #023CB1 !important; /* Blue 700 */
  background-image: none !important; /* Remove gradients */
  color: #FFFFFF; /* White text */
  border-radius: 16px;
  padding: 48px 32px;
}

.hero-title {
  color: #FFFFFF !important;
  font-size: var(--font-size-section-title) !important; /* 32px - same as section-title */
  font-weight: var(--font-weight-bold);
}

.hero-subtitle {
  color: rgba(255, 255, 255, 0.9) !important;
  font-size: var(--font-size-lg);
}

/* Mobile responsive */
@media (max-width: 767px) {
  .hero-title {
    font-size: 24px !important; /* Match section-title mobile */
  }
}

@media (max-width: 480px) {
  .hero-title {
    font-size: 20px !important; /* Match section-title extra small */
  }
}
